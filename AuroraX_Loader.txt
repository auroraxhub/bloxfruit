    while Glow.Parent do
        local t1 = TweenService:Create(Glow, TweenInfo.new(1.2, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {
            ImageTransparency = 0.55,
            Size = UDim2.new(0, 420, 0, 420),

        t1.Completed:Wait()
        if not Glow.Parent then break end

            ImageTransparency = 0.3,
            Size = UDim2.new(0, 380, 0, 380),
        })
        t2:Play()
        t2.Completed:Wait()
    end
end)

    {p = 15,  msg = "Loading modules..."},

    {p = 55,  msg = "Configuring environment..."},
    {p = 78,  msg = "Almost ready..."},
    {p = 95,  msg = "Finalizing..."},
    {p = 100, msg = "Ready!"},
}

local function setProgress(pct, msg)
    local tween = TweenService:Create(BarFill, TweenInfo.new(0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        Size = UDim2.new(pct / 100, 0, 1, 0),
    })
    tween:Play()
    Percent.Text = math.floor(pct) .. "%"
    if msg then Status.Text = msg end
end

for _, stage in ipairs(stages) do
    setProgress(stage.p, stage.msg)
    task.wait(0.55)
end

task.wait(0.3)

--// Fade out

TweenService:Create(BarBg, fadeOut, {BackgroundTransparency = 1}):Play()
TweenService:Create(BarFill, fadeOut, {BackgroundTransparency = 1}):Play()
TweenService:Create(Glow, fadeOut, {ImageTransparency = 1}):Play()

--// Ejecutar el loader principal
local ok, err = pcall(function()

if not ok then
    warn("[Aurora X Hub] Error al cargar el script: " .. tostring(err))
end
